import re


# Funkce pro kontrolu, zda text obsahuje závorky
def has_brackets(text):
    return bool(re.search(r'\(|\)', text))

# Kontrola, zda obsahuje odkazy
if not has_brackets(response):

    response = re.sub(r'\(([^)]+)\)', r'\1', response)
